\frametitle{Our Main Contributions}
\footnotesize{
\begin{itemize}
\item Discrete dynamical systems serve as formal models of diffusion
phenomena in networks (e.g., diffusion of opinions).  \smallskip
%Previous work studied such systems on (general) undirected and directed
%graphs. \smallskip

\item We consider 
\textcolor{blue}{synchronous dynamical systems on directed
acyclic graphs (DAG-SyDSs)}. \smallskip

\item \textbf{Result 1:}~\textcolor{blue}{Reachability} problem for DAG-SyDSs
remains \cpsp-complete even when each local function 
is \textcolor{blue}{symmetric}.\smallskip

\item \textbf{Result 2:}~\textcolor{blue}{Convergence Guarantee} 
problem for DAG-SyDSs
is \ccnp-complete even for DAGs with at most three levels. \smallskip

\item Results 1 and 2 extend those of
\textcolor{blue}{[Chistikov et al., AAAI-2020]} to DAG-SyDSs. \smallskip

\item \textbf{Result 3:}~
\textcolor{blue}{Reachability} problem for DAG-SyDSs
is efficiently solvable when each local function 
is \textcolor{blue}{monotone}.
(For general directed graphs, the problem is \cpsp-complete 
\textcolor{blue}{[Ogihara \& Uchizawa, 2017]}.)
\end{itemize}
